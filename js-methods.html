1 ЧИСЛО
(typeof === 'number')		 1. Math.random(); – возвращает случайное число от 0 до 1:
console.log( Math.random() ); // 0.6357965708679065
2. Math.min(numb1, numb2, ..., numbN); – возвращает минимальное число:
console.log( Math.min(1, 2, 3, -3, 0, -22) ); // -22
3. Math.max(numb1, numb2, ..., numbN); – возвращает максимальное число:
console.log( Math.max(1, 2, 3, -3, 0, -22) ); // 3
4. Math.round(numb); – округляет до ближайшего целого числа:
console.log( Math.round(3.2) ); // 3
console.log( Math.round(-4.7) ); // -5
5. Math.ceil(numb); – округляет в большую сторону:
console.log( Math.ceil(3.2) ); // 4
console.log( Math.ceil(-4.7) ); // -4
6. Math.floor(numb); – округляет в меньшую сторону:
console.log( Math.floor(3.2) ); // 3
console.log( Math.floor(-4.7) ); // -5
7. Math.trunc(numb); – производит удаление дробной части без округления:
console.log( Math.trunc(3.713) ); // 3
8. Math.pow(x, n); – возводит число (x) в степень (n):
console.log( Math.pow(-5, 3) ); // -125
9. Math.sqrt(numb); – возвращает квадратный корень из числа:
console.log( Math.sqrt(81) ); // 9
10. Math.abs(numb); – возвращает модуль числа:
console.log( Math.abs(-25) ); // 25
11.1. e (экспоненциальная запись числа) – умножает число на 1 с указанным количеством нулей после e:
console.log(1000000); // 1000000
console.log(1e6); // 1000000
11.2. e-n (экспоненциальная запись числа) – делит число на 1 с указанным количеством нулей после e:
console.log(0.000001); // 0.000001
console.log(1e-6); // 0.000001
12. .toFixed(numb); – указывает, до скольки знаков после запятой мы хотим округлить число:
let num = 12.12345;

console.log(num.toFixed(2)); // 12.12
13. Number.parseInt(str, radix); – преобразует значение str в целое число по системе счисления radix:
console.log( Number.parseInt('17.1', 10) ); // 17
14. Number.parseFloat(str); – преобразует значение str в число по десятичной системе счисления:
console.log( Number.parseFloat('17.1') ); // 17.1
15. Number(str); – преобразует значение str в число:
let str = '2021';
let num = Number(str);

console.log(num); // 2021
console.log(typeof num === 'number'); // true
16. Number.isNaN(value); – определяет, является ли переданное значение value значением NaN:
console.log( Number.isNaN(123) ); // false
console.log( Number.isNaN(NaN) ); // true
17. Number.isFinite(value); – определяет, является ли переданное значение value конечным числом:
console.log( Number.isFinite(123) ); // true
console.log( Number.isFinite(Infinity) ); // false
18. Number.isInteger(value); – определяет, является ли тип переданного значения value «числом», а само число – целым значением:
console.log( Number.isInteger(123) ); // true
console.log( Number.isInteger(12.3) ); // false

let str = 'String';
let newStr = str.substring(2, 5);

console.log(newStr); // 'rin'

7. .toUpperCase(); – приводит всю строку к верхнему регистру:
let str = 'cAtS';

console.log(str.toUpperCase()); // 'CATS'
8. .concat(string1, string2,... stringN); - объединяет текст из двух или более строк и возвращает новую строку:

console.log(str.trim()); // 'cats'
10. .eval(expression); – интерпритация кода expression в виде строки:

    return eval(str);

11. .indexOf('str'); – ищет часть строки str c её начала, если находит, то возвращает индекс первого символа искомой части, если нет, то выводит -1:
let str = 'cats';

console.log(str.indexOf('cat')); // 0
console.log(str.indexOf('dog')); // -1
12. .lastIndexOf('str'); – ищет часть строки str c её конца, если находит, то возвращает индекс первого символа искомой части, если нет, то выводит -1:
let str = 'cats';

console.log(str.lastIndexOf('cat')); // 0
console.log(str.lastIndexOf('dog')); // -1
13. .startsWith('str', index); – проверяет начинается ли строка с подстроки str, если да, то возвращает true, если нет – false (index – необязательный параметр, который принимает индекс строки, с которого следует начать поиск):
console.log('itIsBeautiful'.startsWith('itIs')); // true
console.log('itIsBeautiful'.startsWith('itis')); // false
console.log('itIsBeautiful'.startsWith('itIs', 0)); // true
14. .endsWith('str', index); – проверяет заканчивается ли строка подстрокой str, если да, то возвращает true, если нет – false (index – необязательный параметр, который принимает индекс строки, с которого следует начать поиск):
console.log('itIsBeautiful'.endsWith('utiful')); // true
console.log('itIsBeautiful'.endsWith('utiful', 0)); // false
console.log('itIsBeautiful'.endsWith('utiful', 13)); // true
15. .includes('str', index); – проверяет содержит ли строка подстроку str, если да, то возвращает true, если нет – false (index – необязательный параметр, который принимает индекс строки, с которого следует начать поиск):
console.log('itIsBeautiful'.includes('itIs')); // true
console.log('itIsBeautiful'.includes('Beau', 0)); // true
console.log('itIsBeautiful'.includes('Beau', 6)); // false
16. String(value); – преобразует значение value в строку:

let arr = [ 1, 2, 3 ];

console.log(typeof String(num) === 'string'); // true -> '25'
console.log(typeof String(bool) === 'string'); // true -> 'true'
console.log(typeof String(arr) === 'string'); // true -> '1,2,3'

4 ОБЪЕКТ
(typeof === 'object')		 1. Object.keys(obj); – показывает все ключи (свойства) объекта obj в виде массива:
const obj = { name: 'ball', color: 'red' };
console.log( Object.keys(obj) ); // [ 'name', 'color' ]

delete obj.color;
console.log( Object.keys(obj) ); // [ 'name' ]
2. Object.values(obj); – показывает все значения объекта obj в виде массива:
const obj = { name: 'ball', color: 'red' };
console.log( Object.values(obj) ); // [ 'ball', 'red' ]

obj.name = 'table';
obj['color'] = 'orange';

console.log( Object.values(obj) ); // [ 'table', 'orange' ]
console.log( Object.keys(obj).map( e => obj[e] ) ); // [ 'table', 'orange' ] - старый способ
3. Object.entries(obj); – показывает все ключи (свойства) объекта obj и их значения в виде массива:
const obj = { name: 'ball', color: 'red' };

console.log( Object.entries(obj) ); // [ [ 'name', 'ball' ], [ 'color', 'red' ] ]
4. for (const key in object) { … } – перебирает все свойства объекта (такие объекты называются итерируемыми), выполняя код с соответствующим свойством на каждом шаге итерации цикла, где key – свойство со значением объекта, а object – объект с перебираемыми элементами:
function getForIn() {
    const object = { a: 1, b: 2, c: 3, d: 4 };
    const newObject = new Object;
    for (const key in object) {
            newObject[key] = object[key];
    }
    return newObject;
}

console.log( getForIn() ); // {a: 1, b: 2, c: 3, d: 4 };
5. 'key' in object – проверяет есть ли свойство key в объекте object:
const obj = { a: 1, b: 2 };

console.log('a' in obj); // true
console.log('c' in obj); // false
6. .hasOwnProperty('key'); – проверяет есть ли свойство key в объекте:
const obj = { name: 'ball', color: 'red' };

console.log( obj.hasOwnProperty('name') ); // true
console.log( obj.hasOwnProperty('surname') ); // false
7. Object.defineProperty(object, 'key', { … } ); – добавляет в объект (object) свойство (key), его значение (value) и даёт возможность настроить это свойство с помощью трёх специальных параметров, которые иначе называются «флаги» (writable, enumerable, configurable):
const obj = { name: 'ball', color: 'red' };

Object.defineProperty(obj, 'age', {
      value: '2',
      writable: true,
      enumerable: true,
      configurable: true
})

console.log(obj); // { name: 'ball', color: 'red', age: '2' }
8. Object.defineProperties(object, { descriptors } ); – добавляет в объект (object) несколько свойств, их значения и даёт возможность настроить эти свойства с помощью трёх специальных параметров, которые иначе называются «флаги» (writable, enumerable, configurable):
const person = new Object;
console.log(person); // {}

Object.defineProperties(person, {
      'name': { value: "John", writable: true, enumerable: true, configurable: true },
      'surname': { value: "Smith", writable: true, enumerable: true, configurable: true  },
      'age': { value: 21, writable: true, enumerable: true, configurable: true }
});

console.log(person); // { name: 'John', surname: 'Smith', age: 21 }
9. Object.getOwnPropertyDescriptor(obj, 'key'); – показывает значение свойства key и другие настройки этого свойства:
const obj = { name: 'ball', color: 'red' };
console.log( Object.getOwnPropertyDescriptor(obj, 'name') );
// { value: 'ball', writable: true, enumerable: true, configurable: true }

const clone = Object.defineProperties( {}, Object.getOwnPropertyDescriptors(obj) );
// данный способ позволяет клонировать объект со всеми его флагами, включая свойства-символы

console.log(clone); // { name: 'ball', color: 'red' }
console.log( Object.getOwnPropertyDescriptor(clone, 'name') );
// { value: 'ball', writable: true, enumerable: true, configurable: true }
10. Object.getOwnPropertySymbols(obj); – показывает все свойства объекта obj с ключами-символами:
const obj = {};
const a = Symbol('a');
const b = Symbol.for('b'); // ищет ранее созданный символ по заданному ключу и, если он найден, возвращает

obj[a] = 'localSymbol';

const objectSymbols = Object.getOwnPropertySymbols(obj);

console.log(objectSymbols); // [ Symbol(a), Symbol(b) ]

          newArr.push(element);

    return newArr;
